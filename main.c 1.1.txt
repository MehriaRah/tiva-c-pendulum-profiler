#include <stdint.h>
#include <stdio.h>
#include "inc/tm4c1294ncpdt.h"

// ============================================================================
// Global Volatile Variables (Safe for Background Interrupts)
// ============================================================================
volatile uint32_t g_echo_cycles = 0;
volatile uint32_t g_new_data_ready = 0; 

// ============================================================================
// STEP 1: Timer 0A ISR (Fires Trigger Pulse every ~62ms)
// ============================================================================
void Timer0A_Handler(void)
{
    TIMER0_ICR_R = 0x01; // Clear Timer 0A timeout flag
    
    // Generate Trigger Pulse on PD0
    GPIO_PORTD_AHB_DATA_R |= (1 << 0);   
    for(volatile int i = 0; i < 20; i++); // Tiny delay for hardware pulse width
    GPIO_PORTD_AHB_DATA_R &= ~(1 << 0);  
}

// ============================================================================
// STEP 3: GPIO Port D ISR (Detects Echo Edges)
// ============================================================================
void GPIO_PortD_Handler(void)
{
    GPIO_PORTD_AHB_ICR_R |= (1 << 1); // Clear Port D Pin 1 flag

    // Check if Echo (PD1) is HIGH (Rising Edge) or LOW (Falling Edge)
    if (GPIO_PORTD_AHB_DATA_R & (1 << 1))
    {
        // --- RISING EDGE: Start Timer 1 Stopwatch ---
        TIMER1_CTL_R &= ~0x01;       // Stop Timer 1 safely to configure
        TIMER1_CFG_R  = 0x00;        // 32-bit configuration
        TIMER1_TAMR_R = 0x02;        // Periodic Countdown Mode
        TIMER1_TAILR_R = 0xFFFFFFFF; // Max capacity start value
        TIMER1_ICR_R  = 0x01;        // Clear old flags
        TIMER1_CTL_R |= 0x01;        // Start counting down!
    }
    else
    {
        // --- FALLING EDGE: Stop Timer 1 & Calculate Ticks ---
        TIMER1_CTL_R &= ~0x01;       // Stop Stopwatch instantly!
        
        uint32_t final_ticks = TIMER1_TAR_R; // Read remaining ticks
        g_echo_cycles = 0xFFFFFFFF - final_ticks; 
        
        g_new_data_ready = 1;        // Notify main() that data is ready
    }
}

// ============================================================================
// Safety Timeout Handler for Timer 1 (Only runs if Echo pin gets stuck)
// ============================================================================
void Timer1A_Handler(void)
{
    TIMER1_ICR_R = 0x01;   // Clear the timeout flag
    TIMER1_CTL_R &= ~0x01; // Shut down the timer
}

// ============================================================================
// Hardware Initialization Function
// ============================================================================
void init_hardware(void)
{
    // 1. Enable Clocks for Port D, Timer 0, and Timer 1
    SYSCTL_RCGCGPIO_R  |= 0x08;  // Port D Clock
    SYSCTL_RCGCTIMER_R |= 0x03;  // Timer 0 (0x01) and Timer 1 (0x02)
    while(!(SYSCTL_PRGPIO_R & 0x08));
    while(!(SYSCTL_PRTIMER_R & 0x03));

    // 2. Setup Port D Basic Digital Direction (PD0=Out, PD1=In)
    GPIO_PORTD_AHB_DEN_R |= (1 << 0) | (1 << 1); 
    GPIO_PORTD_AHB_DIR_R |= (1 << 0);            
    GPIO_PORTD_AHB_DIR_R &= ~(1 << 1);           

    // 3. Configure Timer 0A (Fires Trigger automatically)
    TIMER0_CTL_R  &= ~0x01;       
    TIMER0_CFG_R   = 0x00;         
    TIMER0_TAMR_R  = 0x02;        // Periodic mode
    TIMER0_TAILR_R = 1000000 - 1; // 1,000,000 clock cycles 
    TIMER0_ICR_R   = 0x01;         
    TIMER0_IMR_R  |= 0x01;        // Enable interrupt on timeout

    // 4. Configure Port D Pin 1 for BOTH EDGES Interrupt
    GPIO_PORTD_AHB_IM_R  &= ~(1 << 1); 
    GPIO_PORTD_AHB_IS_R  &= ~(1 << 1); // Edge sensitive
    GPIO_PORTD_AHB_IBE_R |=  (1 << 1); // Interrupt on BOTH edges
    GPIO_PORTD_AHB_ICR_R |=  (1 << 1); 
    GPIO_PORTD_AHB_IM_R  |=  (1 << 1); // Unmask

    // 5. Enable Interrupt Routing inside the Central NVIC Matrix
    NVIC_EN0_R |= (1 << 3);  // Master open gate for GPIO Port D (Interrupt No. 3)
    NVIC_EN0_R |= (1 << 19); // Master open gate for Timer 0A (Interrupt No. 19)
    NVIC_EN0_R |= (1 << 21); // FIXED: Added master open gate for Timer 1A (Interrupt No. 21)

    // 6. Turn the automatic trigger timer ON
    TIMER0_CTL_R |= 0x01;        
}

// ============================================================================
// Main Execution Program (Safe Printing Environment)
// ============================================================================
void main(void)
{
    init_hardware(); 

    while (1)
    {
        // Check if our background workers processed a complete echo pulse
        if (g_new_data_ready == 1)
        {
            g_new_data_ready = 0; // Clear the data flag immediately
            
            // Convert captured cycles to distance
            uint32_t distance_cm = g_echo_cycles / 933; 
            
            // Print safely completely outside the ISR workers
            printf("Distance: %u cm (Cycles: %u)\n", distance_cm, g_echo_cycles);
        }
    }
}
